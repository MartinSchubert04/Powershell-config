import-module terminal-icons

Set-PSReadLineOption -Colors @{
    Default       = "#ffffe3"
    Command       = "#ffffe3"
    Parameter     = "#c0caf5"
    String        = "#9ece6a"
    Operator      = "#89ddff"
    Variable      = "#bb9af7"
    Number        = "#ff9e64"
    Comment       = "#565f89"
    Error         = "#f7768e"
}


Set-PSReadLineKeyHandler -Key "Ctrl+f" `
                         -BriefDescription "AlternarVistaPrediccion" `
                         -LongDescription "Cambia entre vista en l√≠nea y lista para las predicciones de PSReadLine" `
                         -ScriptBlock {
    $options = Get-PSReadLineOption
    if ($options.PredictionViewStyle -eq "InlineView") {
        Set-PSReadLineOption -PredictionViewStyle ListView
    } else {
        Set-PSReadLineOption -PredictionViewStyle InlineView
    }
}
